### Useful info about pynwb patch clamp series ###

## voltage clamp stuff *response* ##
series[0]  # {VoltageClampSeries}
# *stimulus* [less information here, .gain is different though, redundancy]
series[1] # {VoltageClampStimulusSeries}
# important *response* attributes
.capacitance_fast   # {float32} in farads [e-12]
.capacitance_slow   # same as above
.conversion         # {float32} 1e-12 conversion from farads to pF
.rate               # {float32} in Hz
.resistace_comp_bandwidth     # None or ?
.    '       '  correction    # ?
.    '       '  prediction    # ?
.starting_time      # {float64} in s?
.whole_cell_capacitance_comp  # None or ?
.  '     '  series_resistance_comp  # None or ?
.comments # A long string with labnotebook stuff? see .txt files in this dir
## important field from comments for vclamp holding level:
'HS#0:Pipette Offset: 91.79 mV'
'HS#0:V-Clamp Holding Enable: Off'
'HS#0:V-Clamp Holding Level: -69.99 mV'
### also very imporant: only values that change end up in this string
# the comments field should be exactly the same in stim and response series
